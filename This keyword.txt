class Car {
    String brand;
    int year;

    public Car(String brand, int year) {
        this.brand = brand;
        this.year = year;
    }

    public Car() {
        this("Audi", 2024);
    }
}

public class ConstructorDemo {
    public static void main(String[] args) {

        Car myCar = new Car();
        Car myCar2 = new Car("BMW", 2025);

        System.out.println(myCar.brand + " " + myCar.year);
        System.out.println(myCar2.brand + " " + myCar2.year);
    }
}
