class MyThread extends Thread {

    public void run() {

        // Basic run
        System.out.println("Thread is running");

        //Loop + Thread Name
        synchronized (MyThread.class) {
            for (int i = 1; i <= 3; i++) {
                System.out.println(Thread.currentThread().getName());
            }
        }

        //Sleep 
        try {
            Thread.sleep(100);
            System.out.println("Thread resumed after sleep");
        } catch (InterruptedException e) {
            System.out.println("Exception occurred");
        }
    }
}

public class ThreadsDemo {

    public static void main(String[] args) throws InterruptedException {

        //Thread State
        MyThread t1 = new MyThread();
        System.out.println("Before start: " + t1.getState()); // NEW

        t1.start();

        System.out.println("After start: " + t1.getState()); // RUNNABLE

        // Multiple Threads 
        MyThread t2 = new MyThread();
        MyThread t3 = new MyThread();

        t2.start();
        t3.start();

        // join() 
        t1.join();

        System.out.println("Main method is running after t1 completes");
    }
}
